import FolderNode from "./FolderNode";
import { FaPlus } from "react-icons/fa";

export default function FolderTree({
  data = [],
  setData,
  selectedId,
  setSelectedId,
}) {
  if (!Array.isArray(data)) {
    return (
      <p className="text-xl font-bold tracking-wider text-white">
        No folders found
      </p>
    );
  }

  return (
    <div className="h-full flex flex-col">
      {/* Sticky Header */}
      <div className="sticky top-0 z-10 bg-[#1f1c1e] py-4 px-4 border-b border-white/10">
        <div className="w-full flex justify-between items-center">
          <h3 className="text-lg font-bold tracking-widest text-white">
            Explore
          </h3>

          <button
            className="text-sm font-semibold tracking-wider flex justify-center items-center gap-2 text-white hover:text-[#5b5bd6]"
            onClick={() => {
              const name = prompt("Folder name?");
              if (!name) return;

              setData((prev) => [
                ...prev,
                { id: Date.now(), name, children: [], contents: [] },
              ]);
            }}
          >
            <FaPlus className="text-[14px]" />
            <span className="hidden sm:inline">New Folder</span>
          </button>
        </div>
      </div>

      {/* Scrollable Folder List */}
      <div className="flex-1 overflow-y-auto px-4 py-3 space-y-2">
        {data.length === 0 ? (
          <p className="text-sm text-white/60">No folders yet</p>
        ) : (
          data.map((folder) => (
            <FolderNode
              key={folder.id}
              folder={folder}
              setData={setData}
              selectedId={selectedId}
              setSelectedId={setSelectedId}
            />
          ))
        )}
      </div>
    </div>
  );
}
